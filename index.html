// task.js 完全版（全機能統合） const grid = document.getElementById("grid"); const titleInput = document.getElementById("pageTitle"); const selectedCells = new Set(); let history = [], historyIndex = -1; function createGrid() { for (let row = 0; row < 16; row++) { for (let col = 0; col < 20; col++) { const cell = document.createElement("div"); cell.className = "cell"; cell.dataset.row = row; cell.dataset.col = col; const editable = document.createElement("div"); editable.className = "editable"; editable.contentEditable = true; editable.addEventListener("input", saveState); const img = document.createElement("img"); img.style.display = "none"; cell.appendChild(editable); cell.appendChild(img); grid.appendChild(cell); } } } function getCell(row, col) { return [...grid.children].find(c => c.dataset.row == row && c.dataset.col == col); } function saveState() { const state = [...grid.children].map(cell => { const editable = cell.querySelector(".editable"); const img = cell.querySelector("img"); editable.querySelectorAll("input[type='checkbox']").forEach(cb => { cb.checked ? cb.setAttribute("checked", "checked") : cb.removeAttribute("checked"); }); return { row: cell.dataset.row, col: cell.dataset.col, html: editable.innerHTML, fontSize: editable.style.fontSize, bgColor: cell.style.backgroundColor, display: cell.style.display, gridRow: cell.style.gridRow, gridColumn: cell.style.gridColumn, imgSrc: img?.src || '', alignItems: editable.style.alignItems, justifyContent: editable.style.justifyContent, textAlign: editable.style.textAlign }; }); localStorage.setItem("gridState", JSON.stringify(state)); localStorage.setItem("pageTitle", titleInput.value); history = history.slice(0, historyIndex + 1); history.push(JSON.stringify(state)); historyIndex++; } function restoreState() { const saved = localStorage.getItem("gridState"); const savedTitle = localStorage.getItem("pageTitle"); if (savedTitle) titleInput.value = savedTitle; if (!saved) return; const state = JSON.parse(saved); state.forEach(data => { const cell = getCell(data.row, data.col); const editable = cell.querySelector(".editable"); const img = cell.querySelector("img"); editable.innerHTML = data.html || ''; cell.style.display = data.display || 'flex'; editable.querySelectorAll("input[type='checkbox']").forEach(cb => { cb.checked = cb.checked || cb.hasAttribute('checked'); const span = cb.nextElementSibling; span && (span.style.textDecoration = cb.checked ? "line-through" : "none"); cb.addEventListener("change", () => { span && (span.style.textDecoration = cb.checked ? "line-through" : "none"); saveState(); }); }); editable.style.fontSize = data.fontSize || ''; editable.style.alignItems = data.alignItems || 'flex-start'; editable.style.justifyContent = data.justifyContent || 'flex-start'; editable.style.textAlign = data.textAlign || 'left'; cell.style.backgroundColor = data.bgColor || ''; cell.style.display = data.display || 'flex'; cell.style.gridRow = data.gridRow || ''; cell.style.gridColumn = data.gridColumn || ''; if (data.imgSrc) { img.src = data.imgSrc; img.style.display = 'block'; } }); } window.onload = () => { createGrid(); restoreState(); document.querySelectorAll(".cell").forEach(cell => { if (cell.style.gridRow || cell.style.gridColumn) return; if (cell.style.display === 'none') cell.style.display = 'flex'; }); titleInput.addEventListener("input", saveState); const getSelectedCells = () => [...selectedCells].map(k => k.split(',').map(Number)); // 選択機能 let isDragging = false, dragStart = null; grid.addEventListener("mousedown", e => { const cell = e.target.closest(".cell"); if (!cell) return; const row = parseInt(cell.dataset.row), col = parseInt(cell.dataset.col); if (e.shiftKey) { isDragging = true; dragStart = { row, col }; } else { selectedCells.clear(); document.querySelectorAll(".cell").forEach(c => c.classList.remove("selected")); const key = `${row},${col}`; selectedCells.add(key); cell.classList.add("selected"); } }); grid.addEventListener("mousemove", e => { if (!isDragging || !e.shiftKey) return; const cell = e.target.closest(".cell"); if (!cell) return; const endRow = parseInt(cell.dataset.row); const endCol = parseInt(cell.dataset.col); const minRow = Math.min(dragStart.row, endRow), maxRow = Math.max(dragStart.row, endRow); const minCol = Math.min(dragStart.col, endCol), maxCol = Math.max(dragStart.col, endCol); selectedCells.clear(); document.querySelectorAll(".cell").forEach(c => c.classList.remove("selected")); for (let r = minRow; r <= maxRow; r++) { for (let c = minCol; c <= maxCol; c++) { const key = `${r},${c}`; selectedCells.add(key); getCell(r, c).classList.add("selected"); } } }); window.addEventListener("mouseup", () => isDragging = false); // 色変更 document.querySelectorAll(".color-btn").forEach(btn => { btn.addEventListener("click", () => { const color = btn.dataset.color; getSelectedCells().forEach(([r, c]) => getCell(r, c).style.backgroundColor = color); saveState(); }); }); // フォントサイズ変更 document.querySelectorAll(".size-btn").forEach(btn => { btn.addEventListener("click", () => { getSelectedCells().forEach(([r, c]) => getCell(r, c).querySelector(".editable").style.fontSize = btn.dataset.size); saveState(); }); }); // 整列 document.getElementById("alignCenterBtn").addEventListener("click", () => { getSelectedCells().forEach(([r, c]) => { const el = getCell(r, c).querySelector(".editable"); el.style.alignItems = "center"; el.style.justifyContent = "center"; el.style.textAlign = "center"; }); saveState(); }); document.getElementById("alignTopLeftBtn").addEventListener("click", () => { getSelectedCells().forEach(([r, c]) => { const el = getCell(r, c).querySelector(".editable"); el.style.alignItems = "flex-start"; el.style.justifyContent = "center"; el.style.textAlign = "left"; }); saveState(); }); // チェックボックス const insertCheckbox = (bold = false) => { getSelectedCells().forEach(([r, c]) => { const editable = getCell(r, c).querySelector(".editable"); const wrapper = document.createElement("span"); wrapper.style.display = "inline-flex"; wrapper.style.alignItems = "center"; const cb = document.createElement("input"); cb.type = "checkbox"; cb.style.marginRight = "4px"; const span = document.createElement("span"); span.textContent = "項目"; span.contentEditable = true; span.style.outline = "none"; if (bold) span.style.fontWeight = "bold"; cb.addEventListener("change", () => { span.style.textDecoration = cb.checked ? "line-through" : "none"; saveState(); }); wrapper.append(cb, span); editable.appendChild(wrapper); }); saveState(); }; document.getElementById("insertCheckboxBtn").addEventListener("click", () => insertCheckbox(false)); document.getElementById("insertBoldCheckboxBtn").addEventListener("click", () => insertCheckbox(true)); // セル結合 document.getElementById("mergeBtn").addEventListener("click", () => { const coords = getSelectedCells(); if (coords.length < 2) return alert("2つ以上のセルを選択してください。"); const rows = coords.map(c => c[0]), cols = coords.map(c => c[1]); const minR = Math.min(...rows), maxR = Math.max(...rows); const minC = Math.min(...cols), maxC = Math.max(...cols); if ((maxR - minR + 1) * (maxC - minC + 1) !== coords.length) return alert("長方形で選択してください。"); const base = getCell(minR, minC); base.style.gridRow = `${minR + 1} / span ${maxR - minR + 1}`; base.style.gridColumn = `${minC + 1} / span ${maxC - minC + 1}`; coords.forEach(([r, c]) => { if (r !== minR || c !== minC) getCell(r, c).style.display = "none"; }); selectedCells.clear(); document.querySelectorAll(".cell").forEach(c => c.classList.remove("selected")); saveState(); }); // セル分離 document.getElementById("splitBtn").addEventListener("click", () => { if (selectedCells.size !== 1) return alert("1つの結合セルを選択してください。"); const [r, c] = [...selectedCells][0].split(',').map(Number); const base = getCell(r, c); const rowStart = parseInt(base.style.gridRow?.split("/")[0]) - 1 || r; const rowSpan = parseInt(base.style.gridRow?.split("span ")[1]) || 1; const colStart = parseInt(base.style.gridColumn?.split("/")[0]) - 1 || c; const colSpan = parseInt(base.style.gridColumn?.split("span ")[1]) || 1; base.style.gridRow = ""; base.style.gridColumn = ""; for (let i = rowStart; i < rowStart + rowSpan; i++) { for (let j = colStart; j < colStart + colSpan; j++) { getCell(i, j).style.display = "flex"; } } selectedCells.clear(); document.querySelectorAll(".cell").forEach(c => c.classList.remove("selected")); saveState(); }); // 削除 document.getElementById("deleteBtn").addEventListener("click", () => { getSelectedCells().forEach(([r, c]) => { const cell = getCell(r, c); const editable = cell.querySelector(".editable"); editable.innerHTML = ""; // 削除処理にチェックボックスのwrapperも対象 editable.querySelectorAll("span").forEach(span => { if (span.querySelector("input[type='checkbox']")) span.remove(); }); const img = cell.querySelector("img"); img.src = ""; img.style.display = "none"; }); saveState(); }); // リセット document.getElementById("resetBtn").addEventListener("click", () => { localStorage.clear(); location.reload(); }); // Undo / Redo document.getElementById("undoBtn").addEventListener("click", () => { if (historyIndex > 0) { historyIndex--; localStorage.setItem("gridState", history[historyIndex]); restoreState(); } }); document.getElementById("redoBtn").addEventListener("click", () => { if (historyIndex < history.length - 1) { historyIndex++; localStorage.setItem("gridState", history[historyIndex]); restoreState(); } }); };
